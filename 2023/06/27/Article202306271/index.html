class Link:

    def __init__(self, i, j):
        """
        е»әз«Ӣз”ЁжҲ·iдёҺиҠӮзӮ№jзҡ„йҖҡдҝЎй“ҫи·ҜпјҢиҝ”еӣһе…¶д»–е…¬ејҸжүҖйңҖиҰҒзҡ„дҝЎжҒҜ
        :param i: з”ЁжҲ·
        :param j: и®Ўз®—иҠӮзӮ№
        :return: з”ЁдәҺе…¶д»–е…¬ејҸи®Ўз®—зҡ„дҝЎжҒҜ
        """
        self.i = i
        self.j = j
        self.sigma_i_j2 = 1e-13  # иғҢжҷҜеҷӘеЈ° -100dBm
        # дёӨиҖ…д№Ӣй—ҙи·қзҰ»
        self.l_i_j = (abs(i.location[0] - j.location[0]) ** 2 + abs(i.location[1] - j.location[1]) ** 2) ** 0.5
        self.h_i_j = self.l_i_j ** -4  # й“ҫи·Ҝзҡ„дҝЎйҒ“еўһзӣҠ, 4жҳҜи·Ҝеҫ„жҚҹиҖ—жҢҮж•°
        # д»»еҠЎеҚёиҪҪжүҖйңҖиҰҒзҡ„жңҖе°ҸеёҰе®Ҫ
        self.w_i_j = 1e9 * i.P * i.d / (i.delta * i.c * math.log2(1 + i.P * self.h_i_j / self.sigma_i_j2))
        self.b = 5e6  # еӯҗдҝЎйҒ“еёҰе®Ҫ 5MHz
        self.n_i_j = math.ceil(self.w_i_j / self.b)  # жүҖйңҖжңҖе°ҸеӯҗдҝЎйҒ“ж•°зӣ®
        self.g = 2e9  # и®Ўз®—иө„жәҗжңҖе°ҸеҲҶй…ҚеҚ•дҪҚ 2GHz
        # иҠӮзӮ№jеҲҶй…Қз»ҷз”ЁжҲ·iзҡ„жңҖе°Ҹи®Ўз®—иө„жәҗ
        self.f_i_j = i.f_l * i.c / (i.c - i.d * i.f_l / self.w_i_j * math.log2(1 + i.P * self.h_i_j / self.sigma_i_j2))
        self.m_i_j = math.ceil(self.f_i_j / self.g * 100)  # иҜ·жұӮеҚ•дҪҚи®Ўз®—иө„жәҗж•°йҮҸ

        # з”ҹжҲҗжөӢиҜ•ж•°жҚ®
        self.m_i_j = random.randint(12, 18)
        self.n_i_j = random.randint(12, 18)

        self.o_i_j = self.calculate_priority_coefficient()  # з”ЁжҲ·еҜ№жңҚеҠЎиҠӮзӮ№зҡ„дјҳе…Ҳзә§зі»ж•°
        # 4.2.3 и®Ўз®—иҙ№з”ЁйғЁеҲҶ
        # з”ЁжҲ·iеҜ№жңҚеҠЎиҠӮзӮ№jзҡ„иө„жәҗз«һж Үд»·ж ј
        self.rho_i_j = self.i.lamb_c * self.m_i_j + self.i.lamb_w * self.n_i_j
        # жңҚеҠЎиҠӮзӮ№jдёәз”ЁжҲ·iеҲҶй…Қиө„жәҗзҡ„жҲҗжң¬
        self.eta_i_j = self.j.beta_c * self.m_i_j + self.j.beta_w * self.n_i_j
        self.x_i_j = 0  # й»ҳи®ӨдёҚе»әз«Ӣй“ҫжҺҘ

    def calculate_priority_coefficient(self):
        alpha = 0.3  # еҸҜз”Ёж— зәҝз”өиө„жәҗеӣ еӯҗ
        beta = 0.3  # еҸҜз”Ёи®Ўз®—иө„жәҗеӣ еӯҗ
        gamma = 0.4  # и·қзҰ»еӣ еӯҗ
        if alpha + beta + gamma != 1:
            raise Exception("дёҚж»Ўи¶і alpha+beta+gamma = 1 зҡ„еүҚзҪ®иҰҒжұӮ")
        l = (abs(self.i.location[0] - self.j.location[0]) ** 2 + abs(
            self.i.location[1] - self.j.location[1]) ** 2) ** 0.5
        # ж №жҚ®ејҸ(4.17)и®Ўз®—дјҳе…Ҳзә§зі»ж•°
        o_i_j = alpha * self.j.w / self.n_i_j + beta * self.j.c / self.m_i_j + gamma * 1 / l
        return o_i_j

    # и®Ўз®—з”ЁжҲ·iеҜ№иҠӮзӮ№jзҡ„д»·еҖј
    def calculate_u_i_j(self):
        return self.x_i_j * (self.rho_i_j - self.eta_i_j)

def multi_combinatorial_auction(I, J):
    """
    еӨҡиҪ®з»„еҗҲжӢҚеҚ–з®—жі•
    :param I: з”ЁжҲ·зҡ„йӣҶеҗҲ I = {1,2,...,i,...,I} User
    :param J: жңҚеҠЎиҠӮзӮ№йӣҶеҗҲ J={1,2,...,j,...,J}, Node
    :return:
    """
    I_un = I  # жңӘдёӯж Үз”ЁжҲ·
    X = [[0] * len(J) for _ in range(len(I))]  # и®°еҪ•з”ЁжҲ·дёҺиҠӮзӮ№д№Ӣй—ҙзҡ„й“ҫи·ҜзҠ¶жҖҒ
    U_M = 0  # жӢҚеҚ–зҡ„жҖ»ж”¶зӣҠ
    # дёҖдёӘжңҚеҠЎиҠӮзӮ№иЎЁзӨәдёәдёҖиҪ®еҶізӯ–пјҢдёҖе…ұжңүJиҪ®еҶізӯ–
    for N in J:
        # 4.3.1 жҠ•ж ҮжҸҗдәӨйҳ¶ж®ө
        O_s = {}  # еӯҳеӮЁз”ЁжҲ·зҡ„жҠ•ж Үжғ…еҶө
        # з”ЁжҲ·i
        for i in I_un:
            O_s_i = []
            # жүҖжңүзҡ„иҠӮзӮ№j
            for j in J:
                link_i_j = Link(i, j)
                # з”ЁжҲ· i ж №жҚ®ејҸ(4.17)и®Ўз®—еҜ№жңҚеҠЎиҠӮзӮ№зҡ„дјҳе…Ҳзә§зі»ж•° o_i_j
                o_i_j = link_i_j.o_i_j
                O_s_i.append((o_i_j, link_i_j))  # е°Ҷ (o_i_j, link_i_j) дәҢе…ғз»„ж·»еҠ еҲ°еҲ—иЎЁдёӯ
            # еҜ№дјҳе…Ҳзә§иҝӣиЎҢжҺ’еәҸпјҢзЎ®е®ҡжҺ’еәҸйӣҶеҗҲ O_s_iпјҢдҫқж¬Ўеҗ‘ O_s_i дёӯзҡ„жңҚеҠЎиҠӮзӮ№еҸ‘йҖҒеҮәд»·иҜ·жұӮ
            O_s_i = sorted(O_s_i, key=lambda x: x[0], reverse=True)  # жҢүз…§дјҳе…Ҳзә§зі»ж•°иҝӣиЎҢйҷҚеәҸжҺ’еҲ—

            # и®°еҪ•жҲҗеҠҹжҠҘд»·
            j_id = O_s_i[0][1].j.id
            # жЈҖжҹҘиҠӮзӮ№жҳҜеҗҰе·Із»ҸеңЁеӯ—е…ёдёӯпјҢеҰӮжһңдёҚеңЁеҲҷж·»еҠ еҲ°еӯ—е…ёдёӯ
            if j_id not in O_s:
                O_s[j_id] = []
            # е°Ҷз”ЁжҲ·еҲ—иЎЁж·»еҠ еҲ°еҜ№еә”зҡ„иҠӮзӮ№й”®дёӯ
            O_s[j_id].append(O_s_i[0][1])
        # з”ЁжҲ·ж №жҚ®жҺ’еәҸйӣҶеҗҲдёӯзҡ„е…ғзҙ пјҢдҫқж¬Ўеҗ‘е…¶дјҳе…Ҳзә§жңҖй«ҳзҡ„жңҚеҠЎиҠӮзӮ№жҸҗдәӨе…¶еҮәд»·зҡ„еҗ‘йҮҸ
        # иҜ·жұӮжңҚеҠЎиҠӮзӮ№жҸҗдҫӣжңҚеҠЎ

        # и®°еҪ•жҜҸдёҖиҪ®жүҖжңүжңӘдёӯж Үз”ЁжҲ·
        I_temp = []

            # жңҚеҠЎиҠӮзӮ№ j жҺҘж”¶еҲ°з”ЁжҲ·зҡ„иҜ·жұӮдҝЎжҒҜеҗҺеҫ—еҲ°иҜ·жұӮеҲ—иЎЁ I_req_j
            j_id = j.id
            I_req_j = np.array(O_s.get(j_id, None))

            if None in I_req_j:
                continue

            # ж №жҚ®з®—жі• 4.1пјҢеҫ—еҲ°жңҚеҠЎиҠӮзӮ№ж•ҲзӣҠ U_jпјҢжңӘдёӯж Үз”ЁжҲ·йӣҶеҗҲ I_un е’ҢиҝһжҺҘзҹ©йҳө X

                if X_final[i] == "1":
                    j.c = j.c - I_req_j[i].m_i_j  # еҲҶй…Қи®Ўз®—иө„жәҗ
                    j.w = j.w - I_req_j[i].n_i_j  # еҲҶй…ҚйҖҡдҝЎиө„жәҗ

                    I_temp = np.append(I_temp, I_req_j[i].i)
                    X[i][j_id] = 0

            # I_un = np.concatenate((I_un, I_un_j), axis=0)
            # I_un = np.unique(I_un)

        # жӣҙж–°I_un
        I_un = I_temp
        # еҰӮжһңжІЎжңүжңӘдёӯж Үзҡ„з”ЁжҲ·пјҢжҸҗеүҚи·іеҮәжӢҚеҚ–
        if not any(I_un):
            return X, U_M, I_un
        # еӯҳеңЁжңӘдёӯж Үз”ЁжҲ·пјҢжӣҙж–°жңӘдёӯж Үз”ЁжҲ·зҡ„жҠҘд»·
        else:

                # иҝҷйҮҢзҡ„жӣҙж–°ж–№ејҸеёҰжқҘзҡ„еҗҺжһңе°ұжҳҜз”ЁжҲ·зҡ„еҮәд»·иҝҮй«ҳпјҢеҗҺйқўзҡ„ж”¶зӣҠеӨӘеӨ§
                i.lamb_c += i.lamb_c + i.Delta_c * i.lamb_c
                i.lamb_w += i.lamb_w + i.Delta_w * i.lamb_w
    # U_MжҳҜжңҖеҗҺзҡ„жҖ»ж”¶зӣҠ
    return X, U_M, I_un
